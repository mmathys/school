\begin{definition}[mol]
	$1 \ mol = 6.022 \cdot 10^{23}$ Teilchen
\end{definition}

\begin{definition}[n]
	Teilchenanzahl; in $mol$.
\end{definition}

\begin{definition}[m]
	Gewicht absolut; in g.
\end{definition}

\begin{definition}[M]
	Gewicht pro mol; in $g/mol$
\end{definition}

\begin{definition}[V]
	Volumen; in	$l$; $1l=1\ dm^3=0.001\ m^3$
\end{definition}

\begin{definition}[c]
	Konzentration; in $mol/l$
\end{definition}

\begin{definition}[T]
	Konzentration; in $K$
	
	$0\degree C = 273.15K$
\end{definition}

\begin{definition}[p]
	Druck; in $Pa$.
	
	$1\ bar = 10^5 Pa$
	
	Normaldruck: $1\ bar;\ 1013.25\ hPa$
\end{definition}
